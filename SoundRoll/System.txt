# ここでちょっとした設定の変更ができます
# 「=」の右側にある数値や文字列を変更してみてください(指定された値以外は入力しないように！)



# 作業用フォルダの指定をすることが可能です
# 再生したいファイルの入っているディレクトリを指定してください(デフォルトは./music/)
work_directory_location=./music/

# 音量を指定できます 0.0～1.0までの数値を入力してください(デフォルトは0.8)
volume=0.8


# 起動直後に音楽の再生を行うかを指定します
# 0:しない 1:順次に再生 2:シャッフルしてから再生(デフォルトは0)
player_switch = 0


# 一通り再生し終えた後、最初から再生する際にシャッフルをするか指定します
# 順次に再生してる場合は無効となります 0:しない 1:する(デフォルトは1)
always_shuffle = 1
